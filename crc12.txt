#include <stdio.h>
#include <string.h>
int main() {
char data[100],div[14]="1100000001111",temp[14];
int len;
printf("Enter Data bits: ");
scanf("%s",data);
len=strlen(data);
for(int i=0;i<12;i++) data[len+i]='0';
data[len+12]='\0';
strncpy(temp,data,13);
for(int i=0;i<len;i++) {
if(temp[0]=='1') {
for(int j=1;j<13;j++) temp[j]=(temp[j]==div[j])?'0':'1';
}
for(int j=0;j<12;j++) temp[j]=temp[j+1];
temp[12]=data[i+13];
}
printf("CRC Code: %s\n",temp);
return 0;
}
