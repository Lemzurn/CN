#include <stdio.h>
#include <string.h>
int main() {
char str[100];
int size,i;
printf("Enter string: ");
scanf("%s",str);
printf("Enter frame size: ");
scanf("%d",&size);
printf("Result: ");
for(i=0;i<strlen(str);i++) {
if(i>0&&i%size==0) printf(" | ");
printf("%c",str[i]);
}
printf("\n");
return 0;
}
