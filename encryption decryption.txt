#include <stdio.h>
#include <string.h>
int main() {
char str[100];
char key;
int i;
printf("\n--- XOR Encryption/Decryption ---\n");
printf("Enter a string to encrypt/decrypt: ");
fgets(str, sizeof(str), stdin);
printf("Enter a single character key: ");
scanf("%c", &key);
printf("\nOriginal String: %s", str);
for(i = 0; i < strlen(str); i++) {
if(str[i] != '\n') {
str[i] = str[i] ^ key;
}
}
printf("Encrypted/Decrypted String: %s\n", str);
printf("\n--- Verification (Applying key again) ---\n");
for(i = 0; i < strlen(str); i++) {
if(str[i] != '\n') {
str[i] = str[i] ^ key;
}
}
printf("Restored String: %s\n", str);
return 0;
}
